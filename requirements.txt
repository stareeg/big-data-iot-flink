# Зависимости проекта IoT + Flink
#
# Python-пакеты ставятся в виртуальное окружение, созданное именно из Python 3.11
# Создать окружение командой   py -3.11 -m venv .venv   (подробнее в README)
#
# Почему Python 3.11
# PyFlink 2.0 публикует колёса только до cp311, под Python 3.12 и новее их нет
# На хосте по умолчанию стоит Python 3.14, из него окружение не соберётся
#
# Внешние сервисы поднимаются через docker-compose, версии тут для справки
#   Kafka        apache/kafka 3.9.2   режим KRaft, без Zookeeper
#   PostgreSQL   postgres 17
#
# Java нужна для встроенного JVM PyFlink, ставится отдельно, не через pip
#   Flink 2.0 убрал Java 8, подходит JDK 17 или 21, рекомендуется Temurin 17
#
# Jar-коннекторы Flink качаются скриптом scripts/fetch_jars.ps1 в папку jars
#   flink-sql-connector-kafka       4.0.1-2.0
#   flink-connector-jdbc-core       4.0.0-2.0
#   flink-connector-jdbc-postgres   4.0.0-2.0
#   postgresql (драйвер)            42.7.5
# В ветке 2.x JDBC-коннектор разделён на core и postgres, нужны оба jar
# Коннекторы привязаны к минору Flink 2.0 и подходят для всех патчей 2.0.x

# Ядро PyFlink, тянет за собой движок Flink нужной версии и Java-мост
# Берём 2.0.2, последний патч минора 2.0, набор коннекторов под минор 2.0 тот же
apache-flink==2.0.2

# Клиент Kafka для генератора событий
# Канонический kafka-python снова развивается, 2.3.2 работает на Python 3.11 и совместим с брокером 3.9
kafka-python==2.3.2

# Драйвер PostgreSQL для применения ddl и dml из Python и чтения справочника
# Способ интеграции pg уточняется на шаге 4, при инициализации справочника через docker может не понадобиться
psycopg2-binary==2.9.12
